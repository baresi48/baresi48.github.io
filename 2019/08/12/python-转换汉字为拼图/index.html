Python: 转换汉字为拼图

12 Aug, 2019 · Read in about 2 min · (219 Words)
Python Code

代码

#!/usr/bin/python3
# coding=utf-8
__author__ = 'HuaZhen'
__date__ = '2019/8/8'
# 中文转换16*16点阵（依赖HZK16汉字库）

import binascii
from PIL import Image
import random
import os

KEYS = [0x80, 0x40, 0x20, 0x10, 0x08, 0x04, 0x02, 0x01]

#初始化16*16的点阵位置，每个汉字需要16*16=256个点来表示
rect_list = [] * 16
for i in range(16):
    rect_list.append([] * 16)

#具体文字（只支持单个汉字）
text = "爱"

#获取中文的编码
gb2312 = text.encode('gb2312')
hex_str = binascii.b2a_hex(gb2312)
result = str(hex_str, encoding='utf-8')

#根据编码计算文字在汉字库中的位置
area = eval('0x' + result[:2]) - 0xA0
index = eval('0x' + result[2:]) - 0xA0
offset = (94 * (area-1) + (index-1)) * 32
font_rect = None

#读取HZK16汉字库文件中文字数据
with open("HZK16", "rb") as f:
    f.seek(offset)
    font_rect = f.read(32)

#根据读取到HZK中数据给16*16点阵赋值
for k in range(len(font_rect) // 2):
    row_list = rect_list[k]
    for j in range(2):
        for i in range(8):
            asc = font_rect[k * 2 + j]
            flag = asc & KEYS[i]
            row_list.append(flag)

#根据获取到的16*16点阵信息，计算所需图片总数，同时打印点阵到控制台（调试用）
num = 0 
for row in rect_list:
    for i in row:
        if i:
            #前景字符
            print('0', end=' ')
            num += 1
        else:
            #背景字符
            print('.', end=' ')
    print() 

#新建画布，16*16点阵，每个图片边长100
canvas = Image.new('RGB', (1600, 1600), '#FFFFFF')

#指定图片文件夹,从文件夹里取图片
#path="../../../../../tmp/jpg"
path="."
imgs = []
for x in os.listdir(path):
    if x.endswith('jpg'):
        imgs.append(x)
#随机取定量的图片文件
selected_imgs=random.sample(imgs, num)

#在画布上拼接
pr = 0 
for row in range(16):
    for i in range(16):
        if rect_list[row][i]:
            # 取图片
            img = Image.open(selected_imgs[pr])
            pr += 1
            # 缩小图片
            img = img.resize((100, 100), Image.ANTIALIAS)
            # 拼接图片
            canvas.paste(img, (i * 100, row * 100))

#保存图片
canvas.save('../char1.jpg', 'JPEG')
